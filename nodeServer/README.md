<!-- APK 버전 관리 테이블 -->
CREATE TABLE APK_VERSION (
	APK_V INT NOT NULL,
	CREATE_DATE DATE,
	FILE_PATH VARCHAR(100),
	CONSTRAINT APK_VERSION_PK PRIMARY KEY (APK_V)
)
